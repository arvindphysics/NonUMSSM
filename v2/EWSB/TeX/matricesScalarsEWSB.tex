\begin{equation}

\begin{align}

\begin{equation}

\end{equation}

\begin{align}

\tilde{d}_{R,{i \alpha}} = \sum_{j}Z^{D,*}_{j i}\tilde{d}_{{j \alpha}}
\end{align}
